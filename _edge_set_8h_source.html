144  class HalfedgeIterator
145  {
146  public:
148  HalfedgeIterator(Halfedge h = Halfedge(), const EdgeSet* es = nullptr)
149  : m_hnd(h), m_edges(es)
150  {
151  if (m_edges && m_edges->garbage())
152  while (m_edges->isValid(m_hnd) && m_edges->isDeleted(m_hnd))
153  ++m_hnd.m_idx;
154  }
155 
157  Halfedge operator*() const { return m_hnd; }
158 
160  bool operator==(const HalfedgeIterator& rhs) const
161  {
162  return (m_hnd == rhs.m_hnd);
163  }
164 
166  bool operator!=(const HalfedgeIterator& rhs) const
167  {
168  return !operator==(rhs);
169  }
170 
172  HalfedgeIterator& operator++()
173  {
174  ++m_hnd.m_idx;
175  assert(m_edges);
176  while (m_edges->garbage() && m_edges->isValid(m_hnd) &&
177  m_edges->isDeleted(m_hnd))
178  ++m_hnd.m_idx;
179  return *this;
180  }
181 
183  HalfedgeIterator& operator--()
184  {
185  --m_hnd.m_idx;
186  assert(m_edges);
187  while (m_edges->garbage() && m_edges->isValid(m_hnd) &&
188  m_edges->isDeleted(m_hnd))
189  --m_hnd.m_idx;
190  return *this;
191  }
192 
193  private:
194  Halfedge m_hnd;
195  const EdgeSet* m_edges;
196  };
