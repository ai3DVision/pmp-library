93  class VertexIterator
94  {
95  public:
97  VertexIterator(Vertex v = Vertex(), const PointSet* m = NULL)
98  : m_hnd(v), m_ps(m)
99  {
100  if (m_ps && m_ps->garbage())
101  while (m_ps->isValid(m_hnd) && m_ps->isDeleted(m_hnd))
102  ++m_hnd.m_idx;
103  }
104 
106  Vertex operator*() const { return m_hnd; }
107 
109  bool operator==(const VertexIterator& rhs) const
110  {
111  return (m_hnd == rhs.m_hnd);
112  }
113 
115  bool operator!=(const VertexIterator& rhs) const
116  {
117  return !operator==(rhs);
118  }
119 
121  VertexIterator& operator++()
122  {
123  ++m_hnd.m_idx;
124  assert(m_ps);
125  while (m_ps->garbage() && m_ps->isValid(m_hnd) &&
126  m_ps->isDeleted(m_hnd))
127  ++m_hnd.m_idx;
128  return *this;
129  }
130 
132  VertexIterator& operator--()
133  {
134  --m_hnd.m_idx;
135  assert(m_ps);
136  while (m_ps->garbage() && m_ps->isValid(m_hnd) &&
137  m_ps->isDeleted(m_hnd))
138  --m_hnd.m_idx;
139  return *this;
140  }
141 
142  private:
143  Vertex m_hnd;
144  const PointSet* m_ps;
145  };
