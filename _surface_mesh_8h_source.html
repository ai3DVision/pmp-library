151  while (m_mesh->garbage() && m_mesh->isValid(m_hnd) &&

199  class FaceAroundVertexCirculator
200  {
201  public:
203  FaceAroundVertexCirculator(const SurfaceMesh* m = NULL,
204  Vertex v = Vertex())
205  : m_mesh(m), m_active(true)
206  {
207  if (m_mesh)
208  {
209  m_halfedge = m_mesh->halfedge(v);
210  if (m_halfedge.isValid() &&
211  m_mesh->isSurfaceBoundary(m_halfedge))
212  operator++();
213  }
214  }
215 
217  bool operator==(const FaceAroundVertexCirculator& rhs) const
218  {
219  assert(m_mesh);
220  return (m_active && (m_mesh == rhs.m_mesh) &&
221  (m_halfedge == rhs.m_halfedge));
222  }
223 
225  bool operator!=(const FaceAroundVertexCirculator& rhs) const
226  {
227  return !operator==(rhs);
228  }
229 
231  FaceAroundVertexCirculator& operator++()
232  {
233  assert(m_mesh && m_halfedge.isValid());
234  do
235  {
236  m_halfedge = m_mesh->ccwRotatedHalfedge(m_halfedge);
237  } while (m_mesh->isSurfaceBoundary(m_halfedge));
238  m_active = true;
239  return *this;
240  }
241 
243  FaceAroundVertexCirculator& operator--()
244  {
245  assert(m_mesh && m_halfedge.isValid());
246  do
247  m_halfedge = m_mesh->cwRotatedHalfedge(m_halfedge);
248  while (m_mesh->isSurfaceBoundary(m_halfedge));
249  return *this;
250  }
251 
253  Face operator*() const
254  {
255  assert(m_mesh && m_halfedge.isValid());
256  return m_mesh->face(m_halfedge);
257  }
258 
260  operator bool() const { return m_halfedge.isValid(); }
261 
262  // helper for C++11 range-based for-loops
263  FaceAroundVertexCirculator& begin()
264  {
265  m_active = !m_halfedge.isValid();
266  return *this;
267  }
268  // helper for C++11 range-based for-loops
269  FaceAroundVertexCirculator& end()
270  {
271  m_active = true;
272  return *this;
273  }
274

533  bool isTwoManifold(Vertex v) const
534  {
535  // The vertex is non-manifold if more than one gap exists, i.e.
536  // more than one outgoing boundary halfedge.
537  int n(0);
538  HalfedgeAroundVertexCirculator hit = halfedges(v), hend = hit;
539  if (hit)
540  do
541  {
542  if (isSurfaceBoundary(*hit))
543  ++n;
544  } while (++hit != hend);
545  return n < 2;
546  }
